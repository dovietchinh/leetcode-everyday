cmake_minimum_required(VERSION 3.16.1)
project(letcode)
set(THISDIR ${CMAKE_SOURCE_DIR})
file(GLOB SOURCE ${THISDIR}/../source/problem*.cpp)
include_directories(${THISDIR}/../include)
if(${DEBUG}==1)
    message("conpile draft")
    add_executable(draft ${THISDIR}/../source/draft.cpp)
    set_target_properties(draft PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")
else()
    message("conpile problems*.cpp")
    foreach (file ${SOURCE}) 
        get_filename_component(BASENAME ${file} NAME_WLE )
        message(STATUS ${BASENAME})
        add_executable(${BASENAME} ${file})
        set_target_properties(${BASENAME} PROPERTIES RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/bin")
    endforeach()
endif()
